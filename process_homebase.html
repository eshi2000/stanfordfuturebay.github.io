Homebase has asked that we never post counts of locations or users publicly. So for now I’m converting raw numbers to percentages.

summary1 <-
  bay_data %>% 
  left_join(bay_counties %>% mutate(GEOID = as.numeric(GEOID)), by = c("county_code" = "GEOID")) %>% 
  group_by(NAME) %>% 
  summarize(
    Freq = n(),
    avg_hourly_wage_rate = mean(avg_hourly_wage_rate,na.rm=T) %>% round(2),
    total_wages_earned = sum(total_wages_earned, na.rm=T) %>% round(-3),
    hours_worked = mean(hours_worked, na.rm=T) %>% round(1)
  ) %>% 
  ungroup() %>% 
  mutate(
    Freq = (Freq/sum(Freq)*100) %>% round(1)
  )

total_unique_locations <- 
  bay_data %>% 
  pull(location_id) %>% 
  unique() %>% 
  length()

summary_3 <-
  bay_data %>% 
  group_by(event_date) %>% 
  summarize(
    location_count = n()/total_unique_locations
  ) %>% 
  mutate(
    location_count_ma = location_count %>% ma(order = 7)
  )

summary_3 %>% 
  ggplot(
    aes(
      x = event_date,
      y = location_count_ma
    )
  ) +
  geom_line()

total_unique_users <- 
  bay_data %>% 
  pull(user_id) %>% 
  unique() %>% 
  length()

summary_4 <-
  bay_data %>% 
  group_by(event_date) %>% 
  summarize(
    user_count = n()/total_unique_users
  ) %>% 
  mutate(
    user_count_ma = user_count %>% ma(order = 7)
  )

summary_4 %>%
